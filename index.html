plot(t,message.*carrier);

f=linspace(-fs/2,fs/2,l1);
z1=message.*carrier;
z2=fftshift(fft(z1,l1)/l1);
subplot(2,1,2)
plot(f,abs(z2));

figure(3)
subplot(2,1,1)
s=z1.*carrier;
s2=fftshift(fft(s,length(s))/length(s));
plot(f,abs(s2));
hold on
hlp=1./sqrt(1+(f./fm).^(2*10))
plot(f,hlp);

E1=hlp.*s2;
subplot(2,1,2)
plot(f,abs(E1));

figure(4)
d=ifft(ifftshift(E1))*length(E1);
plot(t,2*real(d))

clc; clear; close all; 
N = 1000000; 
a = 1:N; 
x = randn(1,N); 
y = randn(1,N); 
figure(1)
subplot(3,2,1) 
plot(a/1000,x); 
title('In-phase noise'); 
subplot(3,2,2) 
plot(a/1000,y); 
title('Quadrature noise'); 
A = 6.31; 
sigi = A*sin(a/1000); 
sigq = A*cos(a/1000); 
subplot(3,2,3) 
plot(a/1000,sigi); 
title('In-phase signal'); 
subplot(3,2,4) 
plot(a/1000,sigq); 
title('Quadrature signal'); 
subplot(3,2,5) 
plot(a/1000,x+sigi);
title('Noisy In-phase signal'); 
subplot(3,2,6) 
plot(a/1000,y+sigq);
title('Noisy Quadrature-phase signal'); 
c = sqrt(x.^2 + y.^2); 
csigPure = sqrt((sigi).^2 + (sigq).^2); 
csig = sqrt((x+sigi).^2 + (y+sigq).^2); 
edges = 0:0.1:12; 
n = histcounts(c,edges); 
ns_pure = histcounts(csigPure,edges); 
ns = histcounts(csig,edges); 
figure(2) 
plot(edges(1:end-1),n,edges(1:end-1),ns,edges(1:end-1),ns_pure); 
grid on;
legend('Noise','Signal + Noise','Signal'); 
title('Envelope Distribution');

ylabel('Number of Occurrences / Counts');

nfa = find(c > vt);

fprintf('Probability of False Alarm: %f\n', pfa);
fprintf('Probability of Detection: %f\n', pd);

sampleLength=30000;

title('In-phase signal'); 
subplot(3,2,4) 
plot(a(1:sampleLength)/1000,sigq(1:sampleLength)); 
title('Quadrature signal'); 
subplot(3,2,5) 
plot(a(1:sampleLength)/1000,x(1:sampleLength)+sigi(1:sampleLength));
title('Noisy In-phase signal');

title('Noisy Quadrature-phase signal');

PULSE WIDTH MODULATION (PWM) - METHOD 1 (Comparator Method)
clc; clear; close all;
t = 0:0.001:2; 
msg = sin(2*pi*1*t); 
carrier = sawtooth(2*pi*20*t); 

figure(3)
subplot(3,1,1)
plot(t,msg); title('Message Signal');

subplot(3,1,2)
plot(t,carrier); title('Triangular Carrier');

u = zeros(size(t)); 
for i = 1:length(t)
    if msg(i) >= carrier(i) 
        u(i) = 1;
    else
        u(i) = 0;
    end
end

subplot(3,1,3)
plot(t,u);
ylim([-0.1 1.1])
title('PWM Signal');

N = 50; 
demod_pwm = conv(u, ones(1,N)/N, 'same'); 
figure(7)
subplot(2,1,1)
plot(t,u)
title('PWM Signal')
subplot(2,1,2)
plot(t,demod_pwm,'r')
title('Recovered Message from PWM (Simple LPF)')
xlabel('Time'); ylabel('Amplitude');

figure(5)

for i = 1:length(t)
    if msg(i) >= carrier(i)
        u(i) = 1;
    else
        u(i) = 0;
    end
end

subplot(4,1,3)
plot(t,u); title('PWM Signal');
